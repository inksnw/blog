syntax = "proto3";
package services;

//添加option信息
//option go_package = "path;name";
//path 表示生成的go文件的存放地址，会自动生成目录的。
//name 表示生成的go文件所属的包名

option go_package = "./;hello";
message ProdRequest {
  int32 prod_id = 1;
}

message ProdResponse{
  int32 prod_stock = 1;
}
