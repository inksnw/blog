package main

import "fmt"

type User struct {
	Id   int
	Name string
}

type UserAttrFunc func(user *User)
type UserAttrFuncs []UserAttrFunc

func (t UserAttrFuncs) apply(u *User) {
	for _, fun := range t {
		fun(u)
	}
}

func withId(id int) UserAttrFunc {
	return func(user *User) {
		user.Id = id
	}
}

func withName(name string) UserAttrFunc {
	return func(user *User) {
		user.Name = name
	}
}

func NewUser(fs ...UserAttrFunc) *User {
	u := new(User)
	UserAttrFuncs(fs).apply(u)
	return u
}

func main() {
	ins := NewUser(withId(2), withName("xiaoming"))
	fmt.Printf("%+v", ins)
}

	var f func()

		Name string
	}
	var a *ss
	fmt.Println(f, a)

	for _, item := range list {
		if item == nil {
			fmt.Println("nil")
		}

}

	//读正常
	for {
		item, ok := <-ch

package main

type test struct {
}

func (t *test) do(i int) *test {
	fmt.Println(i)
	return t
}

func main() {
	a := test{}
	//defer a.do(1).do(2)
	//a.do(3)
	//输出 1 3 2,defer如果不在闭包函数中,链式调用只有最后一个函数会延时执行

	defer func() {
		defer a.do(1).do(2)
	}()
	a.do(3)
	//输出3,1,2
}

package main

	"fmt"
	"sync"
)

func main() {
	var n int32 = 0
	wg := sync.WaitGroup{}
	for i := 0; i < 1000000; i++ {
		wg.Add(1)
		go func() {
			defer wg.Done()
			//atomic.AddInt32(&n, 1)//输出1000000
			n++ //输出未知
		}()
	}
	wg.Wait()
	fmt.Println(n)
}

package main

import (
	"fmt"
	"sync"
	"time"
)

func job(index int) {
	time.Sleep(time.Millisecond * 500)
	fmt.Printf("执行完毕,序号:%d\n", index)
}

var pool chan struct{}

func main() {
	maxNum := 10
	pool = make(chan struct{}, maxNum)

	wg := sync.WaitGroup{}
	for i := 0; i < 100; i++ {
		pool <- struct{}{} //到达最大长度阻塞
		wg.Add(1)
		go func(index int) {
			defer wg.Done()
			defer func() {
				<-pool
			}()
			job(index)
		}(i)
	}
	wg.Wait()
}

package main

import (
	"fmt"
	"time"
)

//1 .业务过程放到协程
// 2、把业务结果塞入channel
func job() chan string {
	ret := make(chan string)
	go func() {
		time.Sleep(time.Second * 2)
		ret <- "success"
	}()
	return ret
}
func run() (interface{}, error) {
	c := job()
	select {
	case r := <-c:
		return r, nil
	case <-time.After(time.Second * 3):
		return nil, fmt.Errorf("time out")
	}

}
func main() {
	fmt.Println(run())
}

package main

import (
	"fmt"
	"runtime"
	"sync"
)

func main() {
	runtime.GOMAXPROCS(1)
	wg := sync.WaitGroup{}
	for i := 0; i < 5; i++ {
		wg.Add(1)
		go func(sub int) {
			defer wg.Done()
			fmt.Println(sub)
		}(i)
	}
	wg.Wait()
}
//输出 40123 单核下,p默认执行最后一个协程

package main

import (
	"container/list"
	"fmt"
)

func main() {
	data := list.New()

	e8 := data.PushBack(8)
	data.PushBack(9)
	data.PushBack(10)
	data.PushFront(7)

	e85 := data.InsertAfter(8.5, e8)
	data.MoveAfter(e8, e85)
	for e := data.Front(); e != nil; e = e.Next() {
		fmt.Printf("%v ", e.Value)
	}
}
